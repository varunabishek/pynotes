#using args to filter even numbers and put in array
def filter_even(*numbers):
    evens = []
    for x in numbers:
        if x % 2 == 0:
            evens.append(x)
    return evens
result=filter_even(1,2,4,7,65,43,32,90,98)
print(result)

[2, 4, 32, 90, 98]

#sum of n numbers using recursion:

def sum_n(x):
    if x == 0:
        return 1
    else:
        return x + sum_n(x-1)
print(sum_n(5))

16

#factorial by using recursion:

def factorial(x):
    if x == 0:
        return 1
    else:
        return x*factorial(x-1)
print(factorial(7))

5040

#fibbonacci numbers using recursion:

def fibonnaci(x):
    if x <=1:
        return x
    else:
        return fibonnaci(x-1)+fibonnaci(x-2)
print(fibonnaci(8))

21

#factorial using iteration:

def factorial(x):
    ans=1
    if x == 0:
        return x
    else:
        for i in range(1,x+1):
            ans *= i
        return ans
print(factorial(5))
